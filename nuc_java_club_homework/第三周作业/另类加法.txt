package github;
import java.util.Scanner;
public class twonumadd {
	
	 static   int numadd(int a,int b) 
	 {
   	      if(a==0)
   	    	  return b;
   	      else if(b==0)
   	          return a;
   	      else
   	      {
   	    	  int sum  = a^b;
   	    	  int carry = (a&b)<<1;
   	    	 
   	    	  return numadd(sum,carry);
   	      }
   	      
     }
	public static void main(String[] args)
	{
	   Scanner s = new Scanner(System.in);
	   int a = s.nextInt();
	   int b = s.nextInt();
	   System.out.println(numadd(a,b));
	}

}
